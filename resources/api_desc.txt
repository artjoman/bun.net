REST

/party
GET -> retrieves all, partyList.json
POST -> adds new, just status

/party/{partyId}
GET -> get specific, party.json
POST -> update, just status

/party/{partyId}/bun
GET -> retrieves all, bunList.json
POST -> adds new, just status
